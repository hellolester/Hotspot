# This file contains error messages shown to users during HTTP/HTTPS login.
# Messages are friendly and easy to understand.
# Variables used:
#   $(error-orig) - original hotspot error message
#   $(ip)         - client IP address
#   $(username)   - username of client trying to log in

# Internal error
internal-error = Oops! An internal error occurred ($(error-orig)).

# Configuration error
config-error = Configuration error detected ($(error-orig)).

# Not logged in
not-logged-in = You are not logged in (IP: $(ip)).

# IP pool empty
ippool-empty = Cannot assign an IP address – no free addresses available.

# Hotspot shutting down
shutting-down = Hotspot service is currently shutting down.

# User session limit reached
user-session-limit = Voucher/user $(username) is already active.

# License session limit reached
license-session-limit = Your voucher duration has expired ($(error-orig)).

# Wrong MAC for username
wrong-mac-username = Voucher/user $(username) does not match: this MAC address is not yours.

# HTTP CHAP login missing
chap-missing = Login failed! Please enable HTTP PAP in the hotspot server profile.

# Invalid username or password
invalid-username = Voucher/user or password is incorrect. Please try again.

# Invalid MAC address
invalid-mac = Voucher/user $(username) cannot be used from this MAC address.

# Uptime or traffic limit reached
uptime-limit = Voucher/user $(username) has reached the time limit.
traffic-limit = Voucher/user $(username) has reached the data quota.

# RADIUS server timeout
radius-timeout = RADIUS server is not responding. Please try again later.

# Authorization in progress
auth-in-progress = Authorization in progress. Please wait and try again shortly.

# Custom RADIUS error
radius-reply = $(error-orig)
